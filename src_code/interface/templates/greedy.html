def ed_greedy(a,b):
    value=0
    if len(a)>len(b):
        bigger=a ; smaller=b
    else:
        bigger=b ; smaller=a
    BIGGER=[]
    SMALLER=[]
    for i in range(len(bigger)-len(smaller)):
        SMALLER.append(0)

    i=0
    while i<len(bigger):
        BIGGER.append(bigger[i]) ; i+=1
    i=0
    while i<len(smaller):
        SMALLER.append(smaller[i]) ; i+=1


    for i in range(len(BIGGER)):
        if (BIGGER[-1-i])!=(SMALLER[-1-i]):
            value+=1


    return value
